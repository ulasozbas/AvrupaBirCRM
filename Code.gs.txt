function doGet(e) {
  try {
    var action  = String((e&&e.parameter&&e.parameter.action)||"").trim();
    var firmaId = String((e&&e.parameter&&e.parameter.firmaId)||"").trim();
    var tnz     = String((e&&e.parameter&&e.parameter.tnz)||"").trim();

    if (action === "VIPOnayDondurYansisiniBaskan") {
      return HtmlService.createHtmlOutput(musterimTekLifleAnlsOlrAsinaSarsilanTutusM_Mlm(tnz))
        .setTitle("E-Onay Portalı")
        .addMetaTag("viewport","width=device-width, initial-scale=1");
    }
    if (action === "onayla")  return firmaOnayIsle_(firmaId, "ONAY");
    if (action === "reddet")  return firmaOnayIsle_(firmaId, "RED");
    if (e&&e.parameter&&e.parameter.favicon === "1") return faviconPng_();
    if (e&&e.parameter&&e.parameter.view)  return TRACK_viewPage_(String(e.parameter.view||"").trim());
    if (e&&e.parameter&&e.parameter.track) {
      TRACK_markViewed_(String(e.parameter.track||"").trim());
      return ContentService.createTextOutput("").setMimeType(ContentService.MimeType.TEXT);
    }
    if (e&&e.parameter&&e.parameter.page === "ceo") {
      return HtmlService.createHtmlOutputFromFile("ceo_dashboard")
        .setTitle("AvrupaİSG — CEO Komuta Merkezi")
        .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL)
        .addMetaTag("viewport","width=device-width, initial-scale=1");
    }
    var ttT = HtmlService.createTemplateFromFile("Index");
    ttT.USER = { email:"", role:"viewer" };
    return ttT.evaluate()
  .setTitle("AvrupaBirCRM")
  .setSandboxMode(HtmlService.SandboxMode.IFRAME)
  .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL)
  .addMetaTag("viewport","width=device-width, initial-scale=1");
  } catch(err) {
    return HtmlService.createHtmlOutput("HATA: " + (err&&err.message?err.message:err));
  }
}

function AUTH_getUserProfile_(email) {
  function normEmail_(s){ return String(s||"").toLowerCase().trim().replace(/\u00A0/g,"").replace(/\u200B/g,"").replace(/\s+/g,""); }
  email = normEmail_(email);
  if (!email) return null;
  var ss = SpreadsheetApp.openById(CRM_SHEET_ID);
  var sh = ss.getSheetByName("Kullanicilar");
  if (!sh) return null;
  var rows = sh.getDataRange().getDisplayValues();
  for (var i=1; i<rows.length; i++) {
    var e = normEmail_(rows[i][0]);
    if (e !== email) continue;
    var role = String(rows[i][2]||"user").toLowerCase().trim();
    var adSoyad = String(rows[i][3]||"").trim();
    var tel = String(rows[i][4]||"").trim();
    var hazMail = String(rows[i][5]||"").trim() || e;
    var canCost = (role === "admin");
    return { email:e, role:role, adSoyad:adSoyad, tel:tel, hazMail:hazMail, perm:{ canCost:canCost, canSave:true, canMail:true } };
  }
  return null;
}

function AUTH_requireKnownUser_(email) {
  var u = AUTH_getUserProfile_(email);
  if (!u) throw new Error("Yetki yok: Kullanıcı Kullanicilar tablosunda bulunamadı.");
  return u;
}

function AUTH_requirePerm_(email, permKey) {
  var u = AUTH_requireKnownUser_(email);
  if (!u.perm || u.perm[permKey] !== true) throw new Error("Yetki yok: " + permKey);
  return u;
}

function api_loginPin(email, pin) {
  try {
    var temizEmail = String(email||"").toLowerCase().replace(/\s+/g,"").trim();
    var temizPin = String(pin||"").replace(/\s+/g,"").trim();
    if (!temizEmail || !temizPin) return { success:false, basarili:false, msg:"Eksik bilgi" };
    var ss = SpreadsheetApp.openById(CRM_SHEET_ID);
    var sheet = ss.getSheetByName("Kullanicilar");
    if (!sheet) return { success:false, basarili:false, msg:"Kullanıcı tablosu bulunamadı" };
    var data = sheet.getDataRange().getValues();
    for (var i=1; i<data.length; i++) {
      var rowMail = String(data[i][0]||"").toLowerCase().replace(/\s+/g,"").trim();
      var rowPin = String(data[i][1]||"").replace(/\s+/g,"").trim();
      if (rowMail === temizEmail && rowPin === temizPin) {
        var ad = String(data[i][2]||"").trim();
        var izinler = {
          panelFirmalar:data[i][5]===true, panelTeklif:data[i][6]===true, panelFinans:data[i][7]===true,
          panelAktivite:data[i][8]===true, panelPersonel:data[i][9]===true, panelRapor:data[i][10]===true,
          panelTeklifGecmis:data[i][11]===true, panelPerformans:data[i][12]===true, panelDenetim:data[i][13]===true,
          panelTopluMail:data[i][14]===true, panelHarita:data[i][15]===true, panelPatronUyari:data[i][16]===true,
          fnFirmaEkle:data[i][17]===true, fnFirmaDuzenle:data[i][18]===true, fnTeklifOlustur:data[i][19]===true,
          fnTeklifPdf:data[i][20]===true, fnTeklifMail:data[i][21]===true, fnFisEkle:data[i][22]===true,
          fnPersonelEkle:data[i][23]===true, fnPersonelDuzenle:data[i][24]===true, fnIzinOnayla:data[i][25]===true
        };
        var personelSh = ss.getSheetByName("PERSONEL_IK");
        if (personelSh) {
          var pData = personelSh.getDataRange().getValues();
          for (var j=1; j<pData.length; j++) {
            if (String(pData[j][2]||"").toLowerCase() === temizEmail) {
              var jsonStr = String(pData[j][20]||"");
              if (jsonStr && jsonStr.startsWith("{")) { try { izinler = Object.assign(izinler, JSON.parse(jsonStr)); } catch(e) {} }
              break;
            }
          }
        }
        logYaz("SISTEM_GIRIS", "Kullanıcı panele başarıyla giriş yaptı.", "AUTH", "", "", "");
        return { success:true, basarili:true, email:temizEmail, ad:ad, userName:ad, tel:String(data[i][3]||"").trim(), imza:String(data[i][4]||"").trim(), izinler:izinler, pinDegistir: (rowPin === "1234" || String(data[i][26]||"") === "EVET") };
      }
    }
    return { success:false, basarili:false, msg:"Hatalı Giriş" };
  } catch(e) { hataLogla_(e.message, "api_loginPin", email); return { success:false, basarili:false, msg:"Sunucu Hatası: "+e.message }; }
}

function getCEODashboardHTML() {
  try {
    var content = HtmlService.createHtmlOutputFromFile("ceo_dashboard").getContent();
    // Style taglarını al
    var styles = "";
    var styleMatch = content.match(/<style[\s\S]*?<\/style>/gi);
    if (styleMatch) styles = styleMatch.join("\n");
    // Body içeriğini al, script taglarını sil
    var bodyMatch = content.match(/<body[^>]*>([\s\S]*)<\/body>/i);
    var body = bodyMatch ? bodyMatch[1] : content;
    body = body.replace(/<script[\s\S]*?<\/script>/gi, "");
    return styles + body;
  } catch(e) {
    return "<div style='padding:20px;color:red;'>CEO Dashboard yüklenemedi: " + e.message + "</div>";
  }
}

function otomatikYedekle() {
  try {
    var ss = SpreadsheetApp.openById(CRM_SHEET_ID);
    var tarih = Utilities.formatDate(new Date(), "Europe/Istanbul", "dd.MM.yyyy HH:mm");
    var tarihDosya = Utilities.formatDate(new Date(), "Europe/Istanbul", "dd-MM-yyyy");

    // Drive'da klasör bul veya oluştur
    var klasorAdi = "AVRUPABIR CRM YEDEK";
    var klasorler = DriveApp.getFoldersByName(klasorAdi);
    var klasor = klasorler.hasNext() ? klasorler.next() : DriveApp.createFolder(klasorAdi);

    // Script dosyalarını al
    var script = ScriptApp.getScriptId();
    var dosyalar = {
      "CONFIG":    getScriptFileContent_("config"),
      "CODE":      getScriptFileContent_("Code"),
      "FIRMA":     getScriptFileContent_("firma"),
      "TEKLIF":    getScriptFileContent_("teklif"),
      "FINANS":    getScriptFileContent_("finans"),
      "PERSONEL":  getScriptFileContent_("personel"),
      "BORDRO":    getScriptFileContent_("bordro"),
      "RAPOR":     getScriptFileContent_("rapor"),
      "DASHBOARD": getScriptFileContent_("dashboard"),
      "EKSIK":     getScriptFileContent_("eksik")
    };

    // Her dosyayı Drive'a yaz
    var icerik = "// ========== AVRUPABIR CRM YEDEK ==========\n";
    icerik += "// Tarih: " + tarih + "\n";
    icerik += "// ==========================================\n\n";

    Object.keys(dosyalar).forEach(function(ad) {
      if (dosyalar[ad]) {
        icerik += "\n\n// ===== " + ad + " =====\n\n";
        icerik += dosyalar[ad];
      }
    });

    // Eski aynı günlük yedeği sil
    var eskiler = klasor.getFilesByName("CRM_YEDEK_" + tarihDosya + ".gs");
    while (eskiler.hasNext()) eskiler.next().setTrashed(true);

    // Yeni dosyayı yaz
    var blob = Utilities.newBlob(icerik, "text/plain", "CRM_YEDEK_" + tarihDosya + ".gs");
    klasor.createFile(blob);

    Logger.log("✅ Yedek alındı: CRM_YEDEK_" + tarihDosya + ".gs");
    return { ok: true, tarih: tarih };
  } catch(e) {
    Logger.log("❌ Yedek hatası: " + e.message);
    return { ok: false, msg: e.message };
  }
}

function getScriptFileContent_(dosyaAdi) {
  try {
    var url = "https://script.googleapis.com/v1/projects/" + ScriptApp.getScriptId() + "/content";
    var response = UrlFetchApp.fetch(url, {
      headers: { Authorization: "Bearer " + ScriptApp.getOAuthToken() },
      muteHttpExceptions: true
    });
    var data = JSON.parse(response.getContentText());
    var files = data.files || [];
    for (var i = 0; i < files.length; i++) {
      if (files[i].name.toLowerCase() === dosyaAdi.toLowerCase()) {
        return files[i].source || "";
      }
    }
    return "";
  } catch(e) { return ""; }
}

function yedekTriggerKur() {
  // Mevcut yedek triggerlarını sil
  ScriptApp.getProjectTriggers().forEach(function(t) {
    if (t.getHandlerFunction() === "otomatikYedekle") ScriptApp.deleteTrigger(t);
  });
  // Her gece 23:00'de yedek al
  ScriptApp.newTrigger("otomatikYedekle")
    .timeBased()
    .everyDays(1)
    .atHour(23)
    .create();
  Logger.log("✅ Yedek trigger kuruldu — her gece 23:00");
}


function hatirlatmaTriggerkontrol() {
  try {
    var ss = SpreadsheetApp.openById(CRM_SHEET_ID);
    var sh = ss.getSheetByName("Hatirlatmalar");
    if (!sh) return;
    var data = sh.getDataRange().getValues();
    var bugun = new Date();
    bugun.setHours(0,0,0,0);
    for (var i = 1; i < data.length; i++) {
      var row = data[i];
      if (String(row[7]||"").trim() !== "BEKLIYOR") continue;
      var tetik = row[3];
      if (!(tetik instanceof Date)) {
        var p = String(tetik||"").split("-");
        if (p.length === 3) tetik = new Date(p[0], p[1]-1, p[2]);
        else continue;
      }
      tetik.setHours(0,0,0,0);
      if (tetik > bugun) continue;
      var firmaAdi = String(row[2]||"");
      var not = String(row[4]||"");
      var sorumlu = String(row[5]||"");
      var mesaj = "🔔 *Hatırlatma:* " + firmaAdi + "\n📋 " + not + "\n👤 Sorumlu: " + sorumlu;
      chatBildir_(mesaj);
      if (sorumlu && sorumlu.indexOf("@") > -1) {
        MailApp.sendEmail(sorumlu, "Hatırlatma: " + firmaAdi, mesaj);
      }
      sh.getRange(i+1, 8).setValue("GONDERILDI");
    }
  } catch(e) {
    hataLogla_(e.message, "hatirlatmaTriggerkontrol", "sistem");
  }
}

function veritabaniArsivle() {
  try {
    var ss = SpreadsheetApp.openById(CRM_SHEET_ID);
    var yil = new Date().getFullYear() - 1;
    var tablolar = ["FIRMALAR", "TEKLIFLER", "HAREKETLER", "BORDRO", "Hatirlatmalar"];
    var ozet = [];

    tablolar.forEach(function(ad) {
      var sh = ss.getSheetByName(ad);
      if (!sh) return;
      var data = sh.getDataRange().getValues();
      if (data.length < 2) return;

      var arsivAd = ad + "_ARSIV_" + yil;
      var arsivSh = ss.getSheetByName(arsivAd);
      if (!arsivSh) {
        arsivSh = ss.insertSheet(arsivAd);
        arsivSh.appendRow(data[0]);
      }

      var tasınan = [];
      var kalan = [data[0]];

      for (var i = 1; i < data.length; i++) {
        var row = data[i];
        var tarih = row[0];
        var tarihYil = null;
        if (tarih instanceof Date) {
          tarihYil = tarih.getFullYear();
        } else {
          var str = String(tarih || "");
          var m = str.match(/(\d{4})/);
          if (m) tarihYil = parseInt(m[1]);
        }
        if (tarihYil && tarihYil <= yil) {
          tasınan.push(row);
        } else {
          kalan.push(row);
        }
      }

      if (tasınan.length > 0) {
        arsivSh.getRange(arsivSh.getLastRow() + 1, 1, tasınan.length, tasınan[0].length).setValues(tasınan);
        sh.clearContents();
        sh.getRange(1, 1, kalan.length, kalan[0].length).setValues(kalan);
        ozet.push(ad + ": " + tasınan.length + " satır arşivlendi");
      }
    });

    Logger.log("Arşivleme tamamlandı:\n" + ozet.join("\n"));
    chatBildir_("🗄️ *Yıllık Arşivleme Tamamlandı*\n" + ozet.join("\n"));
    return { ok: true, ozet: ozet };
  } catch(e) {
    hataLogla_(e.message, "veritabaniArsivle", "sistem");
    return { ok: false, msg: e.message };
  }
}

function api_churnAnaliziYap() {
  try {
    var ss = SpreadsheetApp.openById(CRM_SHEET_ID);
    var bugun = new Date();
    bugun.setHours(0,0,0,0);

    // FIRMALAR
    var firmaSh = ss.getSheetByName("FIRMALAR");
    if (!firmaSh) return { ok: false, msg: "FIRMALAR bulunamadı" };
    var firmaData = firmaSh.getDataRange().getValues();
    var firmaHead = firmaData[0];
    var firmaIdx = {};
    firmaHead.forEach(function(h,i){ firmaIdx[String(h||"").trim()] = i; });

    // HAREKETLER
    var harSh = ss.getSheetByName("HAREKETLER");
    var harData = harSh ? harSh.getDataRange().getValues() : [];

    // ISG_PLANLAMA
    var isgSh = ss.getSheetByName("ISG_PLANLAMA");
    var isgData = isgSh ? isgSh.getDataRange().getValues() : [];

    // TEKLIFLER
    var tekSh = ss.getSheetByName("TEKLIFLER");
    var tekData = tekSh ? tekSh.getDataRange().getValues() : [];

    // Her firma için son aktivite tarihini bul
    var firmaAktivite = {};
    for (var i = 1; i < harData.length; i++) {
      var fId = String(harData[i][2]||"").trim();
      var tarih = harData[i][1];
      if (!fId || !tarih) continue;
      if (!firmaAktivite[fId] || tarih > firmaAktivite[fId]) firmaAktivite[fId] = tarih;
    }

    // Her firma için son ziyaret tarihini bul
    var firmaZiyaret = {};
    for (var i = 1; i < isgData.length; i++) {
      var fId = String(isgData[i][2]||"").trim();
      var tarih = isgData[i][1];
      if (!fId || !tarih) continue;
      if (!firmaZiyaret[fId] || tarih > firmaZiyaret[fId]) firmaZiyaret[fId] = tarih;
    }

    // Her firma için bekleyen teklif var mı
    var firmaTeklif = {};
    for (var i = 1; i < tekData.length; i++) {
      var fId = String(tekData[i][0]||"").trim();
      var durum = String(tekData[i][tekData[0].indexOf("Durum")]||"").trim();
      var tarih = tekData[i][tekData[0].indexOf("Tarih")];
      if (!fId) continue;
      if (durum === "BEKLIYOR" && tarih) {
        var fark = Math.floor((bugun - new Date(tarih)) / 86400000);
        if (fark > 30) firmaTeklif[fId] = fark;
      }
    }

    var riskliListe = [];

    for (var i = 1; i < firmaData.length; i++) {
      var row = firmaData[i];
      var firmaId = String(row[0]||"").trim();
      var firmaAdi = String(row[1]||"").trim();
      var sorumlu = String(row[6]||"").trim();
      if (!firmaId || !firmaAdi) continue;

      var puan = 0;
      var sebepler = [];

      // Son aktivite
      var sonAkt = firmaAktivite[firmaId];
      if (!sonAkt) {
        puan += 2;
        sebepler.push("Hiç aktivite yok");
      } else {
        var aktFark = Math.floor((bugun - new Date(sonAkt)) / 86400000);
        if (aktFark > 90) { puan += 2; sebepler.push("Son aktivite " + aktFark + " gün önce"); }
        else if (aktFark > 60) { puan += 1; sebepler.push("Son aktivite " + aktFark + " gün önce"); }
      }

      // Son ziyaret
      var sonZiy = firmaZiyaret[firmaId];
      if (!sonZiy) {
        puan += 2;
        sebepler.push("Hiç saha ziyareti yok");
      } else {
        var ziyFark = Math.floor((bugun - new Date(sonZiy)) / 86400000);
        if (ziyFark > 60) { puan += 2; sebepler.push("Son ziyaret " + ziyFark + " gün önce"); }
      }

      // Bekleyen teklif
      if (firmaTeklif[firmaId]) {
        puan += 1;
        sebepler.push("Teklif " + firmaTeklif[firmaId] + " gündür bekliyor");
      }

      // Takip durumu
      var takipSonuc = String(row[17]||"").trim();
      if (takipSonuc === "OLUMSUZ") { puan += 3; sebepler.push("Takip olumsuz işaretlendi"); }

      if (puan >= 3) {
        riskliListe.push({
          firmaId:  firmaId,
          firmaAdi: firmaAdi,
          sorumlu:  sorumlu,
          puan:     puan,
          risk:     puan >= 5 ? "YUKSEK" : "ORTA",
          sebepler: sebepler.join(", ")
        });
      }
    }

    riskliListe.sort(function(a,b){ return b.puan - a.puan; });

    if (riskliListe.length > 0) {
      var mesaj = "⚠️ *Churn Analizi — Risk Listesi*\n";
      riskliListe.slice(0, 5).forEach(function(f) {
        mesaj += "\n🔴 *" + f.firmaAdi + "* (" + f.risk + ")\n" + f.sebepler + "\n";
      });
      chatBildir_(mesaj);
    }

    return { ok: true, liste: riskliListe };
  } catch(e) {
    hataLogla_(e.message, "api_churnAnaliziYap", "sistem");
    return { ok: false, msg: e.message };
  }
}

function api_ceoHaftaOzeti() {
  try {
    var ss = SpreadsheetApp.openById(CRM_SHEET_ID);
    var bugun = new Date();
    bugun.setHours(23,59,59,0);
    var haftaBasi = new Date(bugun);
    haftaBasi.setDate(haftaBasi.getDate() - 7);
    haftaBasi.setHours(0,0,0,0);

    // Yeni firmalar
    var firmaSh = ss.getSheetByName("FIRMALAR");
    var yeniFirma = 0;
    if (firmaSh) {
      var fData = firmaSh.getDataRange().getValues();
      var fHead = fData[0];
      var fTarihIdx = fHead.indexOf("OlusturmaTarihi") > -1 ? fHead.indexOf("OlusturmaTarihi") : 13;
      for (var i=1; i<fData.length; i++) {
        var t = fData[i][fTarihIdx];
        if (t instanceof Date && t >= haftaBasi && t <= bugun) yeniFirma++;
      }
    }

    // Teklifler
    var tekSh = ss.getSheetByName("TEKLIFLER");
    var teklifSayisi = 0, teklifTutar = 0;
    if (tekSh) {
      var tData = tekSh.getDataRange().getValues();
      var tHead = tData[0];
      var tTarihIdx = tHead.indexOf("Tarih") > -1 ? tHead.indexOf("Tarih") : 2;
      var tTutarIdx = tHead.indexOf("GenelToplam") > -1 ? tHead.indexOf("GenelToplam") : 8;
      for (var i=1; i<tData.length; i++) {
        var t = tData[i][tTarihIdx];
        if (t instanceof Date && t >= haftaBasi && t <= bugun) {
          teklifSayisi++;
          teklifTutar += Number(tData[i][tTutarIdx]||0);
        }
      }
    }

    // Saha ziyaretleri
    var ziyaretSayisi = 0;
    var zSh = ss.getSheetByName("ISG_ZIYARET_KANIT");
    if (zSh) {
      var zData = zSh.getDataRange().getValues();
      for (var i=1; i<zData.length; i++) {
        var t = zData[i][1];
        if (t instanceof Date && t >= haftaBasi && t <= bugun) ziyaretSayisi++;
        else {
          var str = String(t||"");
          var p = str.split(".");
          if (p.length === 3) {
            var d = new Date(p[2].substring(0,4), p[1]-1, p[0]);
            if (d >= haftaBasi && d <= bugun) ziyaretSayisi++;
          }
        }
      }
    }

    // Churn
    var churnSonuc = api_churnAnaliziYap();
    var riskliSayi = churnSonuc.ok ? churnSonuc.liste.length : 0;
    var riskliListe = churnSonuc.ok ? churnSonuc.liste.slice(0,3).map(function(f){ return f.firmaAdi; }).join(", ") : "";

    // Bekleyen teklifler
    var bekleyenSayisi = 0;
    if (tekSh) {
      var tData2 = tekSh.getDataRange().getValues();
      var tHead2 = tData2[0];
      var tDurumIdx = tHead2.indexOf("Durum") > -1 ? tHead2.indexOf("Durum") : 6;
      for (var i=1; i<tData2.length; i++) {
        if (String(tData2[i][tDurumIdx]||"").trim() === "BEKLIYOR") bekleyenSayisi++;
      }
    }

    // Gemini özeti
    var apiKey = PropertiesService.getScriptProperties().getProperty("GEMINI_API_KEY");
    var prompt = "Aşağıdaki iş haftalık verileri için Türkçe, kısa ve akıcı bir yönetici özeti yaz (max 3 cümle). " +
      "Olumlu gelişmeleri vurgula, dikkat edilmesi gerekenleri belirt.\n\n" +
      "Veriler:\n" +
      "- Yeni eklenen firma: " + yeniFirma + "\n" +
      "- Gönderilen teklif: " + teklifSayisi + " adet, toplam " + teklifTutar.toLocaleString("tr-TR") + " TL\n" +
      "- Saha ziyareti: " + ziyaretSayisi + "\n" +
      "- Bekleyen teklif: " + bekleyenSayisi + "\n" +
      "- Risk altındaki firma: " + riskliSayi + (riskliListe ? " (" + riskliListe + ")" : "") + "\n";

    var payload = { contents: [{ parts: [{ text: prompt }] }] };
    var response = UrlFetchApp.fetch(
      "https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent?key=" + apiKey,
      { method: "POST", contentType: "application/json", payload: JSON.stringify(payload), muteHttpExceptions: true }
    );
    var json = JSON.parse(response.getContentText());
    var ozet = json.candidates && json.candidates[0] ? json.candidates[0].content.parts[0].text : "Özet alınamadı.";

    return {
      ok: true,
      yeniFirma: yeniFirma,
      teklifSayisi: teklifSayisi,
      teklifTutar: teklifTutar,
      ziyaretSayisi: ziyaretSayisi,
      bekleyenSayisi: bekleyenSayisi,
      riskliSayi: riskliSayi,
      ozet: ozet
    };
  } catch(e) {
    hataLogla_(e.message, "api_ceoHaftaOzeti", "ceo");
    return { ok: false, msg: e.message };
  }
}

function kodYedekle_(sadeceIndex) {
  try {
    var folderId = "1qpkcPzPpRJZXeQ7gOVYdH1sIhUuB8CFX";
    var folder = DriveApp.getFolderById(folderId);
    var tarih = Utilities.formatDate(new Date(), "Europe/Istanbul", "yyyyMMdd_HHmm");

    var proje = ScriptApp.getScriptId();
    var url = "https://script.googleapis.com/v1/projects/" + proje + "/content";
    var resp = UrlFetchApp.fetch(url, {
      headers: { Authorization: "Bearer " + ScriptApp.getOAuthToken() },
      muteHttpExceptions: true
    });
    var json = JSON.parse(resp.getContentText());
    var files = json.files || [];

    var subFolders = folder.getFolders();
    var yedekKlasor = null;
    while (subFolders.hasNext()) {
      var f = subFolders.next();
      if (f.getName().indexOf("YEDEK_") === 0) yedekKlasor = f;
    }
    if (!yedekKlasor || !sadeceIndex) {
      yedekKlasor = folder.createFolder("YEDEK_" + tarih);
    }

    files.forEach(function(f) {
      if (!f.source) return;

      var uzanti = f.type === "JSON" ? ".json" : f.type === "HTML" ? ".html" : ".gs";

      if (f.name === "Index") {
        // Index'i 3 parçaya böl
        var src = f.source;
        var parcaBoyutu = Math.ceil(src.length / 3);
        for (var p = 0; p < 3; p++) {
          var parca = src.substring(p * parcaBoyutu, (p + 1) * parcaBoyutu);
          var doc = DocumentApp.create("Index_PARCA" + (p + 1) + ".html");
          doc.getBody().setText(parca);
          DriveApp.getFileById(doc.getId()).moveTo(yedekKlasor);
          Logger.log("Index parça " + (p + 1) + " kaydedildi.");
        }
        return;
      }

      var doc = DocumentApp.create(f.name + uzanti);
      doc.getBody().setText(f.source);
      DriveApp.getFileById(doc.getId()).moveTo(yedekKlasor);
      Logger.log(f.name + " kaydedildi.");
    });

    Logger.log("Tamamlandı.");
    return { ok: true };
  } catch(e) {
    Logger.log("Hata: " + e.message);
    return { ok: false, msg: e.message };
  }
}

      Utilities.formatDate(new Date(), "GMT+3", "dd.MM.yyyy HH:mm"), // TarihSaat
      islem,                                        // İşlem
      entity,                                       // Entity
      entityId                                      // EntityId
    ];
    
    sheet.appendRow(rowData);
  } catch (e) {
    console.error("Log hatası: " + e.toString());
  }
}

function tumKodlariYedekle() {
  try {
    var scriptId = ScriptApp.getScriptId();
    var url = "https://script.googleapis.com/v1/projects/" + scriptId + "/content";
    
    var response = UrlFetchApp.fetch(url, {
      headers: {
        Authorization: "Bearer " + ScriptApp.getOAuthToken()
      }
    });
    
    var content = JSON.parse(response.getContentText());
    var files = content.files || [];
    
    // Ana klasör oluştur
    var tarih = Utilities.formatDate(new Date(), "Europe/Istanbul", "ddMMyyyy_HHmm");
    var klasorAdi = "CRM_YEDEK_" + tarih;
    var klasor = DriveApp.createFolder(klasorAdi);
    
    var MAX_BOYUT = 500000; // 500KB - her parça max bu kadar
    var ozetMetin = "=== CRM YEDEK ÖZET ===\n";
    ozetMetin += "Tarih: " + Utilities.formatDate(new Date(), "Europe/Istanbul", "dd.MM.yyyy HH:mm:ss") + "\n";
    ozetMetin += "Toplam Dosya: " + files.length + "\n\n";
    
    files.forEach(function(file) {
      var dosyaAdi = file.name;
      var kaynak = file.source || "";
      var tip = file.type || "";
      
      ozetMetin += "- " + dosyaAdi + " (" + tip + ") - " + kaynak.length + " karakter\n";
      
      // Dosya küçükse tek parça kaydet
      if (kaynak.length <= MAX_BOYUT) {
        var icerik = "DOSYA: " + dosyaAdi + "\n";
        icerik += "TIP: " + tip + "\n";
        icerik += "TARIH: " + Utilities.formatDate(new Date(), "Europe/Istanbul", "dd.MM.yyyy HH:mm:ss") + "\n";
        icerik += "========================================\n\n";
        icerik += kaynak;
        
        klasor.createFile(dosyaAdi + ".txt", icerik, MimeType.PLAIN_TEXT);
        Logger.log("✅ Kaydedildi: " + dosyaAdi);
      } 
      // Dosya büyükse parçalara böl
      else {
        var toplamParca = Math.ceil(kaynak.length / MAX_BOYUT);
        Logger.log("⚠️ Büyük dosya: " + dosyaAdi + " → " + toplamParca + " parçaya bölünüyor");
        
        for (var p = 0; p < toplamParca; p++) {
          var baslangic = p * MAX_BOYUT;
          var bitis = Math.min(baslangic + MAX_BOYUT, kaynak.length);
          var parca = kaynak.substring(baslangic, bitis);
          
          var parcaIcerik = "DOSYA: " + dosyaAdi + "\n";
          parcaIcerik += "TIP: " + tip + "\n";
          parcaIcerik += "PARCA: " + (p + 1) + "/" + toplamParca + "\n";
          parcaIcerik += "TARIH: " + Utilities.formatDate(new Date(), "Europe/Istanbul", "dd.MM.yyyy HH:mm:ss") + "\n";
          parcaIcerik += "========================================\n\n";
          parcaIcerik += parca;
          
          var parcaDosyaAdi = dosyaAdi + "_PARCA" + (p + 1) + "_" + toplamParca + ".txt";
          klasor.createFile(parcaDosyaAdi, parcaIcerik, MimeType.PLAIN_TEXT);
          Logger.log("✅ Parça kaydedildi: " + parcaDosyaAdi);
        }
        
        ozetMetin += "  → " + toplamParca + " parçaya bölündü\n";
      }
    });
    
    // Özet dosyasını kaydet
    klasor.createFile("_OZET.txt", ozetMetin, MimeType.PLAIN_TEXT);
    
    Logger.log("✅ Tüm yedekler tamamlandı!");
    Logger.log("Klasör URL: " + klasor.getUrl());
    
    return { 
      ok: true, 
      url: klasor.getUrl(), 
      klasor: klasorAdi,
      dosyaSayisi: files.length
    };
    
  } catch(e) {
    Logger.log("❌ Hata: " + e.message);
    return { ok: false, msg: e.message };
  }
}

  google.script.run
    .withSuccessHandler(function(res) {
      if (res && res.ok) {

        toplantiListeYukle();
      } else {

    })
    .withFailureHandler(function(e) { alert('Hata: ' + e.message); })
    .api_toplantiCevapla({ id: id, karar: karar, email: window.currentUserEmail });
}

function api_toplantiCevapla(params) {
  try {
    var id = String(params.id || "").trim();
    var karar = String(params.karar || "").trim();
    var email = String(params.email || "").toLowerCase().trim();
    
    if (!id || !karar) return { ok: false, msg: "Eksik parametre" };
    
    // --- 1. TAKVİM GÜNCELLEME VE MAİL GÖNDERME BÖLÜMÜ ---
    try {
      var event = CalendarApp.getEventById(id);
      if (event) {
        // Takvim durumunu güncelle
        if (karar === 'kabul') {
          event.setMyStatus(CalendarApp.GuestStatus.YES);
        } else {
          event.setMyStatus(CalendarApp.GuestStatus.NO);
        }

        // Organizatöre mail gönder
        var organizator = event.getCreators()[0];
        var baslik = event.getTitle();
        var gonderen = Session.getActiveUser().getEmail();
        var mesaj = "Sayın Yetkili,\n\n" + gonderen + " kullanıcısı '" + baslik + 
                    "' konulu toplantı davetinize " + 
                    (karar === 'kabul' ? "KABUL (Geleceğim)" : "RED (Gelemeyeceğim)") + 
                    " yanıtı vermiştir.\n\nBilgilerinize.";

        MailApp.sendEmail(organizator, "Toplantı Katılım Durumu: " + baslik, mesaj);
      }
    } catch (e) {
      console.error("Takvim/Mail hatası (İşleme devam ediliyor): " + e.toString());
      // Takvimde bulunamazsa bile aşağıdaki Sheet güncellemesi devam etsin diye hatayı yutuyoruz.
    }

    // --- 2. MEVCUT SHEET GÜNCELLEME BÖLÜMÜ ---
    var ss = SpreadsheetApp.openById(CRM_SHEET_ID);
    var sh = ss.getSheetByName("Toplantılar");
    if (!sh) return { ok: false, msg: "Sheet bulunamadı" };
    
    var data = sh.getDataRange().getValues();
    var satir = -1;
    
    for (var i = 1; i < data.length; i++) {
      if (String(data[i][0]) === id) {
        satir = i + 1;
        break;
      }
    }
    
    if (satir < 0) return { ok: false, msg: "Toplantı bulunamadı" };
    
    sh.getRange(satir, 12).setValue(karar);
    sh.getRange(satir, 14).setValue(new Date().toLocaleString('tr-TR'));
    
    var renk = karar === 'kabul' ? '#dcfce7' : '#fee2e2';
    sh.getRange(satir, 1, 1, 14).setBackground(renk);
    
    return { ok: true };
  } catch(e) {
    return { ok: false, msg: e.message };
  }
}

function api_teklifDurumGuncelle(id, yeniDurum) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getSheetByName("TEKLIFLER") || ss.getSheets()[0]; 
    var data = sheet.getDataRange().getValues();
    
    for (var i = 1; i < data.length; i++) {
      // id burada "1001" gibi bir değerdir. data[i][0] da A sütunudur.
      if (String(data[i][0]).trim() === String(id).trim()) { 
        // F Sütunu (Durum) 6. sıradadır.
        sheet.getRange(i + 1, 6).setValue(yeniDurum || "ONAYLANDI"); 
        return { ok: true };
      }
    }
    return { ok: false, msg: "ID bulunamadı: " + id };
  } catch (e) {
    return { ok: false, msg: e.toString() };
  }
}
