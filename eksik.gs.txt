function api_isgPlanListeGetir(tarihVeyaP, email, rol) {
  try {
    var p = {};
    if (typeof tarihVeyaP === "object" && tarihVeyaP !== null) {
      p = tarihVeyaP;
    } else {
      p = { tarih: tarihVeyaP, email: email, rol: rol };
    }
    var ss = SpreadsheetApp.openById(CRM_SHEET_ID);
    var sh = ss.getSheetByName("ISG_PLANLAMA");
    if (!sh || sh.getLastRow() < 2) return { ok: true, liste: [] };
    var data = sh.getDataRange().getDisplayValues();
    var head = data[0];
    var idx = {};
    head.forEach(function(h,i){ idx[String(h||"").trim()] = i; });
    var liste = [];
    for (var i=1; i<data.length; i++) {
      var row = data[i];
      if (!row[0]) continue;
      var tarih = row[1];
      var tarihStr = tarih instanceof Date ? Utilities.formatDate(tarih,"Europe/Istanbul","dd.MM.yyyy") : String(tarih||"");
      if (p.tarih) {
        var aranan = String(p.tarih).trim();
        // ISO formatını TR formatına çevir (2026-03-16 → 16.03.2026)
        if (aranan.indexOf("-") > -1) {
          var prc = aranan.split("-");
          if (prc.length === 3) aranan = prc[2] + "." + prc[1] + "." + prc[0];
        }
        if (tarihStr !== aranan) continue;
      }
      if (p.personel && String(row[4]||"").toLowerCase() !== p.personel.toLowerCase()) continue;
      if (p.email && String(row[4]||"").toLowerCase() !== String(p.email||"").toLowerCase() && String(row[10]||"").toLowerCase() !== String(p.email||"").toLowerCase()) continue;
      liste.push({ planId:String(row[0]||""), tarih:tarihStr, firmaId:String(row[2]||""), firmaAdi:String(row[3]||""), firma:String(row[3]||""), personelEmail:String(row[4]||""), personel:String(row[4]||""), personelAd:String(row[5]||""), hizmetTipi:String(row[6]||""), hizmet:String(row[6]||""), planlananDk:Number(row[7]||0), sure:String(row[7]||""), durum:String(row[8]||""), notlar:String(row[9]||""), planlananSaat: (function() {
  var s = row[12];
  if (!s) return "";
  if (s instanceof Date) return Utilities.formatDate(s, "Europe/Istanbul", "HH:mm");
  return String(s).trim();
})() });
    }
    liste.sort(function(a,b){ return a.tarih < b.tarih ? -1 : 1; });
    return liste;
  } catch(e) { return { ok: false, msg: e.message, liste: [] }; }
}

function api_isgCokluPlanKaydet(p) {
  try {
    p = p || {};
    var planlar = [];
    if (Array.isArray(p)) {
      planlar = p;
    } else if (p && Array.isArray(p.planlar)) {
      planlar = p.planlar;
    } else if (p && p.tarih) {
      planlar = [p];
    }
    var ilkPlan = planlar[0] || {};
    var eklenen = 0;

    for (var i=0; i<planlar.length; i++) {
      planlar[i]._coklu = true;
      var res = api_isgPlanKaydet(planlar[i]);
      if (res.ok) eklenen++;
    }
    if (eklenen > 0) {
      var ilk = ilkPlan;
      sahaChatBildir_("📅 *Yeni Plan Eklendi*\nFirma: " + String(ilk.firmaAdi||ilk.firma||"") + "\nPersonel: " + String(ilk.personelEmail||ilk.personel||"").split("@")[0].toUpperCase() + "\nTarih: " + String(ilk.tarih||"") + "\nSaat: " + String(ilk.planlananSaat||"--:--") + "\nHizmet: " + String(ilk.hizmetTipi||ilk.hizmet||"") + "\nToplam: " + eklenen + " plan");
    }
    return { ok: true, eklenen: eklenen };
  } catch(e) { return { ok: false, msg: e.message }; }
}

function api_isgPlanIptal(planId) {
  try {
    var ss = SpreadsheetApp.openById(CRM_SHEET_ID);
    var sh = ss.getSheetByName("ISG_PLANLAMA");
    if (!sh) return { ok: false, msg: "ISG_PLANLAMA yok" };
    var data = sh.getDataRange().getValues();
    for (var i=1; i<data.length; i++) {
      if (String(data[i][0]) === String(planId)) {

        return { ok: true };
      }
    }
    return { ok: false, msg: "Plan bulunamadı" };
  } catch(e) { return { ok: false, msg: e.message }; }
}

function api_kasaKapat(p) {
  try {
    p = p || {};
    var ss = SpreadsheetApp.openById(CRM_SHEET_ID);
    var sh = ss.getSheetByName("FİNANS_CARİ");
    if (!sh) return { ok: false, msg: "FİNANS_CARİ yok" };

    var bugun = Utilities.formatDate(new Date(), "Europe/Istanbul", "dd.MM.yyyy");
    var data = sh.getDataRange().getValues();
    var islemler = [];
    var toplamGelir = 0, toplamGider = 0;

    for (var i = 1; i < data.length; i++) {
      var row = data[i];
      var id = String(row[0]||"").trim();
      if (!id || (!id.startsWith("FIN_") && !id.startsWith("CRI_"))) continue;

      var tarihStr = "";
      try {
        var d = new Date(row[1]);
        if (!isNaN(d.getTime())) tarihStr = Utilities.formatDate(d, "Europe/Istanbul", "dd.MM.yyyy");
      } catch(e) {}
      if (tarihStr !== bugun) continue;

      var tip = String(row[2]||"").toUpperCase();
      var tutar = parseFloat(row[6])||0;
      var saat = "";
      try { saat = Utilities.formatDate(new Date(row[1]), "Europe/Istanbul", "HH:mm"); } catch(e) {}

      var islem = {
        saat: saat,
        tip: tip,
        odemeTipi: String(row[4]||"-"),
        firmaAdi: String(row[5]||""),
        aciklama: String(row[5]||""),
        tutar: tutar
      };

      islemler.push(islem);
      if (tip === "GELİR" || tip === "TAHSİLAT") toplamGelir += tutar;
      else if (tip === "GİDER" || tip === "BORÇ") toplamGider += tutar;
    }

    // Kasa bakiyesi
    var shKasa = ss.getSheetByName("FİNANS_KASA");
    var sonBakiye = 0;
    if (shKasa && shKasa.getLastRow() > 1) {
      var kData = shKasa.getDataRange().getValues();
      for (var k=1; k<kData.length; k++) sonBakiye += parseFloat(kData[k][3]||0)||0;
    }

    return {
      ok: true,
      islemler: islemler,
      ozet: {
        toplamGelir: toplamGelir,
        toplamGider: toplamGider,
        sonBakiye: sonBakiye
      }
    };
  } catch(e) { return { ok: false, msg: e.message }; }
}

function api_bankoVerileriGetir() {
  try {
    var ss = SpreadsheetApp.openById(CRM_SHEET_ID);
    var sh = ss.getSheetByName("FİNANS_CARİ");
    if (!sh) return { ok: false, msg: "FİNANS_CARİ yok" };

    var bugun = Utilities.formatDate(new Date(), "Europe/Istanbul", "dd.MM.yyyy");
    var data = sh.getDataRange().getValues();
    var gelirler = [], giderler = [], kargolar = [];
    var nakitToplam = 0, kkartToplam = 0, cariToplam = 0, havaleToplam = 0;

    for (var i = 1; i < data.length; i++) {
      var row = data[i];
      // ID yoksa veya bozuksa atla
      var id = String(row[0]||"").trim();
      if (!id || (!id.startsWith("FIN_") && !id.startsWith("CRI_") && !id.startsWith("CARİ_") && !id.startsWith("CARI_"))) continue;
      
      var tarihStr = "";
      try {
        var d = new Date(row[1]);
        if (!isNaN(d.getTime())) tarihStr = Utilities.formatDate(d, "Europe/Istanbul", "dd.MM.yyyy");
      } catch(e) {}
      if (tarihStr !== bugun) continue;

      var tip    = String(row[2]||"").toUpperCase();
      var kasa   = String(row[4]||"").toUpperCase();
      var tutar  = parseFloat(row[6])||0;
      var aciklama = String(row[5]||"");
      var saat = "";
      try { saat = Utilities.formatDate(new Date(row[1]), "Europe/Istanbul", "HH:mm"); } catch(e) {}

      var islem = { id:id, tarih:tarihStr, saat:saat, tip:tip, kasa:kasa, aciklama:aciklama, tutar:tutar, kaydeden:String(row[11]||""), durum:String(row[12]||"") };

      if (tip === "GELİR" || tip === "TAHSİLAT") {
        gelirler.push(islem);
        if (kasa === "NAKİT") nakitToplam += tutar;
        else if (kasa.indexOf("KREDİ") > -1 || kasa.indexOf("KREDI") > -1 || kasa.indexOf("POS") > -1) kkartToplam += tutar;
        else if (kasa.indexOf("CARİ") > -1 || kasa.indexOf("CARI") > -1) cariToplam += tutar;
        else if (kasa.indexOf("HAVALE") > -1 || kasa.indexOf("EFT") > -1 || kasa.indexOf("BANKA") > -1) havaleToplam += tutar;
      } else if (tip === "GİDER" || tip === "BORÇ") {
        giderler.push(islem);
      } else if (tip === "KARGO") {
        kargolar.push(islem);
      }
    }
    // Bekleyen kargolar - MOBIL_SAHA'dan
    var shMobil = ss.getSheetByName("MOBIL_SAHA");
    bekleyenKargolar = [];
    if (shMobil && shMobil.getLastRow() > 1) {
      var mobilData = shMobil.getDataRange().getValues();
      for (var m = 1; m < mobilData.length; m++) {
        var mSonuc = String(mobilData[m][5]||"");
        var mTeslimat = String(mobilData[m][6]||"");
        // Sonuç hazır ama kargo verilmemiş
        if (mSonuc === "Sonuçlar Hazır" && mTeslimat !== "Kargoya Verildi" && mTeslimat !== "Elden Teslim Edildi") {
          var kTarih = mobilData[m][3];
var kTarihStr = kTarih instanceof Date ? Utilities.formatDate(kTarih,"Europe/Istanbul","dd.MM.yyyy") : String(kTarih||"").split("T")[0];
var kFirmaAdi = String(mobilData[m][2]||"").trim();
var kTarihStr2 = kTarih instanceof Date ? 
  Utilities.formatDate(kTarih,"Europe/Istanbul","dd.MM.yyyy") : 
  String(kTarih||"").substring(0,10).split("-").reverse().join(".");

bekleyenKargolar.push({
  satirNo: m + 1,
  firmaAdi: kFirmaAdi || "Bilinmiyor",
  tarih: kTarihStr2,
  tel: "",
  sonuc: mSonuc,
  teslimat: mTeslimat
});
        
        }
      }
    }
    return { 
      ok: true, 
      islemler: gelirler.concat(giderler),
      gelirler: gelirler,
      giderler: giderler,
      bekleyenKargolar: bekleyenKargolar,
      ozet: { nakit: nakitToplam, kkart: kkartToplam, cari: cariToplam, havale: havaleToplam, toplam: nakitToplam + kkartToplam + cariToplam + havaleToplam }
    };
  } catch(e) { return { ok: false, msg: e.message }; }
}

          if (!isNaN(bit) && bit < bugun) continue;
        }
        var adSoyad = String(kData[k][ki["Görevlendirilen Kişi Ad Soyad"]]||"").trim();
        var sure    = Number(kData[k][ki["Çalışma Süresi"]]||0);
        if (adSoyad && sure > 0) {
          personelDk[adSoyad] = (personelDk[adSoyad]||0) + sure;
        }
      }
    }

    liste.forEach(function(p) {
      var atanmis = personelDk[p.adSoyad] || 0;
      p.atanmisDk = atanmis;
      p.bosDk     = Math.max(0, LIMIT - atanmis);
    });

    return { ok: true, liste: liste };
  } catch(e) {
    return { ok: false, liste: [], msg: e.message };
  }
}

// ==========================================
function sabahGunlukProgram() {
  var ss = SpreadsheetApp.openById(CRM_SHEET_ID);
  var bugun = new Date();
  var bugunStr = Utilities.formatDate(bugun, "Europe/Istanbul", "dd.MM.yyyy");
  var gunAdi = ["Pazar","Pazartesi","Salı","Çarşamba","Perşembe","Cuma","Cumartesi"][bugun.getDay()];

  var shPlan = ss.getSheetByName("ISG_PLANLAMA");
  if (!shPlan || shPlan.getLastRow() < 2) return;
  var planVals = shPlan.getDataRange().getValues();

  var personelPlanlar = {};
  var personelSira = [];

  for (var i = 1; i < planVals.length; i++) {
    var row = planVals[i];
    var tarih = row[1];
    var tarihStr = tarih instanceof Date ? Utilities.formatDate(tarih, "Europe/Istanbul", "dd.MM.yyyy") : "";
    if (tarihStr !== bugunStr) continue;
    if (String(row[8]||"").trim() === "İPTAL" || String(row[8]||"").trim() === "IPTAL") continue;
    var personelEmail = String(row[4]||"").trim();
    var personelAd = String(row[5]||"").trim();
    if (!personelAd || personelAd === personelEmail) personelAd = personelEmail.split("@")[0].replace("."," ").toUpperCase();
    var firma = String(row[3]||"").trim();
    var hizmet = String(row[6]||"").trim();
    var planSaat = row[12] instanceof Date ? Utilities.formatDate(row[12], "Europe/Istanbul", "HH:mm") : String(row[12]||"--:--").trim();
    if (!personelPlanlar[personelEmail]) {
      personelPlanlar[personelEmail] = { ad: personelAd || personelEmail.split("@")[0], planlar: [] };
      personelSira.push(personelEmail);
    }
    personelPlanlar[personelEmail].planlar.push({ firma: firma, hizmet: hizmet, saat: planSaat });
  }

  // Başlık mesajı
  sahaChatBildir_("🌅 GÜNAYDIN! " + gunAdi + " " + bugunStr + "\nBugünün saha programı aşağıda:");

  if (personelSira.length === 0) {
    sahaChatBildir_("📭 Bugün için plan bulunmuyor.");
    return;
  }

  // Her personel için ayrı mesaj
  for (var p = 0; p < personelSira.length; p++) {
    var pData = personelPlanlar[personelSira[p]];
    var mesaj = "👤 " + pData.ad + "\n─────────────";
    for (var k = 0; k < pData.planlar.length; k++) {
      var pl = pData.planlar[k];
      mesaj += "\n🕐 " + pl.saat + "\n🏢 " + pl.firma + "\n🔧 " + pl.hizmet + "\n";
    }
    sahaChatBildir_(mesaj);
    Utilities.sleep(1000);
  }

  sahaChatBildir_("💪 İyi çalışmalar!");
  chatBildir_("📅 Sabah planları saha ekibine gönderildi. Toplam: " + personelSira.length + " personel");
}

// ==========================================
function sahaUyariGonder_() {
  var ss = SpreadsheetApp.openById(CRM_SHEET_ID);
  var bugun = new Date();
  var bugunStr = Utilities.formatDate(bugun, "Europe/Istanbul", "dd.MM.yyyy");
  var simdi = bugun.getHours() * 60 + bugun.getMinutes();

  var shPlan = ss.getSheetByName("ISG_PLANLAMA");
  if (!shPlan || shPlan.getLastRow() < 2) return;
  var planVals = shPlan.getDataRange().getValues();

  for (var i = 1; i < planVals.length; i++) {
    var row = planVals[i];
    var tarih = row[1];
    var tarihStr = tarih instanceof Date ? Utilities.formatDate(tarih, "Europe/Istanbul", "dd.MM.yyyy") : "";
    if (tarihStr !== bugunStr) continue;
    var durum = String(row[8]||"").trim();
    if (durum === "TAMAMLANDI" || durum === "İPTAL" || durum === "DEVAM_EDIYOR") continue;
    var planSaatRaw = row[12];
    var planSaat = planSaatRaw instanceof Date ? Utilities.formatDate(planSaatRaw, "Europe/Istanbul", "HH:mm") : String(planSaatRaw||"").trim();
    if (!planSaat) continue;
    var saatParts = planSaat.match(/(\d+):(\d+)/);
    if (!saatParts) continue;
    var planDk = parseInt(saatParts[1]) * 60 + parseInt(saatParts[2]);
    var fark = planDk - simdi;
    var birSaatOnce = fark >= 55 && fark <= 65;
    var tamSaatinde = fark >= -5 && fark <= 5;
    if (!birSaatOnce && !tamSaatinde) continue;

    var personelAd = String(row[5]||row[4]||"").trim();
    var firmaAdi = String(row[3]||"").trim();
    var hizmet = String(row[6]||"").trim();

    var mesaj = tamSaatinde ?
      "🚨 ZİYARET VAKTİ!\n\n👤 Uzman: " + personelAd + "\n🏢 Firma: " + firmaAdi + "\n📋 Hizmet: " + hizmet + "\n🕐 Saat: " + planSaat + "\n\n⏩ Hemen hareket edin!" :
      "⏰ ZİYARET HATIRLATMASI\n\n👤 Uzman: " + personelAd + "\n🏢 Firma: " + firmaAdi + "\n📋 Hizmet: " + hizmet + "\n🕐 Saat: " + planSaat + "\n\n🔔 1 saat sonra ziyaret var!";

    sahaChatBildir_(mesaj);

    // Yönetim kanalına da ilet (sadece saatinde)
    if (tamSaatinde) {
      chatBildir_("🚨 ZİYARET VAKTİ — " + personelAd + " → " + firmaAdi + " (" + planSaat + ")");
    }
  }
}

function sahaTetikleyicileriKur() {
  // Mevcut tetikleyicileri temizle
  var triggers = ScriptApp.getProjectTriggers();
  triggers.forEach(function(t) {
    if (t.getHandlerFunction() === "sabahGunlukProgram" || t.getHandlerFunction() === "sahaUyariGonder_") {
      ScriptApp.deleteTrigger(t);
    }
  });

  // Sabah 08:00'de günlük program
  ScriptApp.newTrigger("sabahGunlukProgram")
    .timeBased()
    .everyDays(1)
    .atHour(8)
    .inTimezone("Europe/Istanbul")
    .create();

  // Her 5 dakikada bir hatırlatma/uyarı
  ScriptApp.newTrigger("sahaUyariGonder_")
    .timeBased()
    .everyMinutes(5)
    .create();

  Logger.log("Tetikleyiciler kuruldu.");
}

function api_isgUzmanOner(p) {
  try {
    p = p || {};
    var tehlike = String(p.tehlikeSinifi || "").trim();
    var ss = SpreadsheetApp.openById(CRM_SHEET_ID);

    // Personel listesini al
    var res = api_personelUnvanlaraGore();
    var liste = res.liste || [];

    // Tehlike sınıfına göre uzman sınıfı filtresi
    var uzmanSiniflar = [];
    if (tehlike === "Az Tehlikeli")  uzmanSiniflar = ["A","B","C"];
    if (tehlike === "Tehlikeli")     uzmanSiniflar = ["B","A"];
    if (tehlike === "Çok Tehlikeli") uzmanSiniflar = ["A"];

    // Uzmanları filtrele
   function normUnvan(s) {
      return String(s||"").toUpperCase()
        .replace(/İ/g,"I").replace(/Ş/g,"S").replace(/Ç/g,"C")
        .replace(/Ö/g,"O").replace(/Ü/g,"U").replace(/Ğ/g,"G");
    }

    var uzmanlar = liste.filter(function(p) {
      var unvan = normUnvan(p.unvan);
      if (unvan.indexOf("ISG UZMANI") === -1 && unvan.indexOf("IS GUVENLIGI") === -1) return false;
      if (uzmanSiniflar.length === 0) return true;
      for (var i=0; i<uzmanSiniflar.length; i++) {
        if (unvan.indexOf(uzmanSiniflar[i] + " SINIFI") > -1 || unvan.indexOf(uzmanSiniflar[i] + " SINIF") > -1) return true;
      }
      return false;
    }).sort(function(a,b){ return b.bosDk - a.bosDk; });

    var hekimler = liste.filter(function(p) {
      var unvan = normUnvan(p.unvan);
      return unvan.indexOf("ISYERI HEKIMI") > -1 || unvan.indexOf("HEKIM") > -1;
    }).sort(function(a,b){ return b.bosDk - a.bosDk; });

    var dspLer = [];
    if (tehlike === "Çok Tehlikeli") {
      dspLer = liste.filter(function(p) {
        var unvan = normUnvan(p.unvan);
        return unvan.indexOf("DSP") > -1 || unvan.indexOf("HEMSIRE") > -1;
      }).sort(function(a,b){ return b.bosDk - a.bosDk; });
    }

    return {
      ok: true,
      onerilen: {
        uzman:  uzmanlar[0]  || null,
        hekim:  hekimler[0]  || null,
        dsp:    dspLer[0]    || null
      },
      uzmanlar:  uzmanlar,
      hekimler:  hekimler,
      dspLer:    dspLer
    };
  } catch(e) { return { ok: false, msg: e.message }; }
}

function api_katipVerisiIsleSheete() {
  try {
    var ss = SpreadsheetApp.openById(CRM_SHEET_ID);
    var shKatip = ss.getSheetByName("ISG_KATIP");
    var shISG   = ss.getSheetByName("ISG_HIZMETLER");
    if (!shKatip || !shISG) return { ok: false, msg: "Sheet bulunamadı" };

    // ISG_KATIP başlıkları
    var kData = shKatip.getDataRange().getValues();
    var kHead = kData[0];
    var ki = {};
    kHead.forEach(function(h,i){ ki[String(h||"").trim()] = i; });

    // ISG_HIZMETLER başlıkları
    var iData = shISG.getDataRange().getValues();
    var iHead = iData[0];
    var ii = {};
    iHead.forEach(function(h,i){ ii[String(h||"").trim()] = i; });

    // ISG_HIZMETLER'de eksik sütunları ekle
    var eklenecek = ["SGKSicilNo","CalisanSayisi","TehlikeSinifi"];
    var lc = shISG.getLastColumn();
    eklenecek.forEach(function(col){
      if (ii[col] == null) {
        shISG.getRange(1, lc+1).setValue(col);
        ii[col] = lc;
        lc++;
      }
    });

    // Katip verisini firma adına göre haritala
    var katipMap = {}; // firmaAdi.toLowerCase() → {sgk, calisan, tehlike}
    var bugun = new Date();
    for (var k=1; k<kData.length; k++) {
      var bitStr = String(kData[k][ki["Sözleşme Bitiş Tarihi"]]||"");
      if (bitStr) {
        var bit = new Date(bitStr);
        if (!isNaN(bit) && bit < bugun) continue; // süresi dolmuş
      }

function api_firmaEksikDoldur() {
  try {
    var ss = SpreadsheetApp.openById(CRM_SHEET_ID);
    var shFirma  = ss.getSheetByName("FİRMALAR") || ss.getSheetByName("FIRMALAR");
    var shKatip  = ss.getSheetByName("ISG_KATIP");
    if (!shFirma) return { ok: false, msg: "FIRMALAR sheet bulunamadı" };

    var fData = shFirma.getDataRange().getValues();
    var fHead = fData[0].map(function(h){ return String(h||"").trim(); });
    var fi = {};
    fHead.forEach(function(h,i){ fi[h] = i; });

    // Eksik sütunları ekle
    var eklenecek = ["FirmaAdresi","Telefon","Sehir","Enlem","Boylam"];
    var lc = shFirma.getLastColumn();
    eklenecek.forEach(function(col){
      if (fi[col] == null) {
        shFirma.getRange(1, lc+1).setValue(col);
        fi[col] = lc;
        lc++;
      }
    });

    // ISG_KATIP'ten il bilgisi haritası
    var katipMap = {};
    if (shKatip && shKatip.getLastRow() > 1) {
      var kData = shKatip.getDataRange().getValues();
      var kHead = kData[0];
      var ki = {};
      kHead.forEach(function(h,i){ ki[String(h||"").trim()] = i; });
      for (var k=1; k<kData.length; k++) {
        var firmaAdi = String(kData[k][ki["Hizmet Alan İşyeri Unvanı"]]||"").trim().toLowerCase();
        var il       = String(kData[k][ki["Hizmet Alan İşyeri İli"]]||"").trim();
        var sgk      = String(kData[k][ki["Hizmet Alan İşyeri SGK/DETSİS No"]]||"").trim();
        if (firmaAdi && il) katipMap[firmaAdi] = { il: il, sgk: sgk };
      }
    }

    // İl → koordinat haritası (Türkiye illeri)
    var ilKoord = {
      "İSTANBUL":{"lat":41.0082,"lng":28.9784},"ANKARA":{"lat":39.9208,"lng":32.8541},
      "İZMİR":{"lat":38.4237,"lng":27.1428},"BURSA":{"lat":40.1826,"lng":29.0665},
      "ANTALYA":{"lat":36.8969,"lng":30.7133},"ADANA":{"lat":37.0000,"lng":35.3213},
      "KOCAELİ":{"lat":40.8533,"lng":29.8815},"KONYA":{"lat":37.8746,"lng":32.4932},
      "GAZİANTEP":{"lat":37.0662,"lng":37.3833},"MERSİN":{"lat":36.8000,"lng":34.6333}
    };

    var guncellenen = 0;
    for (var r=1; r<fData.length; r++) {
      var row = fData[r];
      var firmaAdi = String(row[fi["FirmaAdi"]||1]||"").trim().toLowerCase();
      if (!firmaAdi) continue;

      var katip = katipMap[firmaAdi];
      if (!katip) continue;

      // Şehir — boşsa doldur
      if (katip.il && !String(row[fi["Sehir"]]||"").trim()) {
        shFirma.getRange(r+1, fi["Sehir"]+1).setValue(katip.il);
      }

      // Koordinat — boşsa il merkezini yaz
      var enlem = row[fi["Enlem"]];
      var boylam = row[fi["Boylam"]];
      if ((!enlem || !boylam) && ilKoord[katip.il.toUpperCase()]) {
        var koord = ilKoord[katip.il.toUpperCase()];
        shFirma.getRange(r+1, fi["Enlem"]+1).setValue(koord.lat);
        shFirma.getRange(r+1, fi["Boylam"]+1).setValue(koord.lng);
      }

      guncellenen++;
    }

    return { ok: true, msg: guncellenen + " firma güncellendi" };
  } catch(e) {
    return { ok: false, msg: e.message };
  }
}

    var isgSh = ss.getSheetByName("ISG_HIZMETLER");
    var isgData = isgSh && isgSh.getLastRow() > 1 ? isgSh.getDataRange().getValues() : [];

function chatMesajGonder(mesaj, hedef) {
  try {
    var ss = SpreadsheetApp.openById(CRM_SHEET_ID);
    var sh = ss.getSheetByName("AYAR");
    var data = sh.getDataRange().getValues();
    var webhookUrl = "";
    var anahtar = hedef === "grup" ? "CHAT_WEBHOOK_GRUP" : "CHAT_WEBHOOK_MUDUR";
    for (var i = 0; i < data.length; i++) {
      if (String(data[i][0]).trim() === anahtar) { webhookUrl = String(data[i][1]).trim(); break; }
    }
    if (!webhookUrl) return false;
    var payload = JSON.stringify({ text: mesaj });
    UrlFetchApp.fetch(webhookUrl, {
      method: "post",
      contentType: "application/json",
      payload: payload
    });
    return true;
  } catch(e) { Logger.log("Chat hata: " + e.message); return false; }
}

function gorevGecikmeKontrol() {
  try {
    var ss = SpreadsheetApp.openById(CRM_SHEET_ID);
    var sh = ss.getSheetByName("ISG_GOREV_HAVUZU");
    if (!sh || sh.getLastRow() < 2) return;
    var data = sh.getDataRange().getValues();
    var header = data[0];
    var idx = {};
    header.forEach(function(h, i) { idx[String(h||"").trim()] = i; });
    var bugun = new Date();
    bugun.setHours(0,0,0,0);
    var gecikmisList = [];
    var bugunList = [];
    for (var i = 1; i < data.length; i++) {
      var row = data[i];
      var durum = String(row[idx["Durum"]||6]||"").toUpperCase();
      if (durum === "TAMAMLANDI" || durum === "IPTAL") continue;
      var sonTarih = new Date(row[idx["SonTarih"]||4]);
      sonTarih.setHours(0,0,0,0);
      var fark = Math.ceil((sonTarih - bugun) / (1000*60*60*24));
      var gorevAdi = String(row[idx["GorevBaslik"]||2]||"");
      var personel = String(row[idx["AtananKisi"]||3]||"");
      if (fark < 0) {
        gecikmisList.push("🔴 " + personel + " — " + gorevAdi + " (" + Math.abs(fark) + " gün gecikmiş)");
      } else if (fark === 0) {
        bugunList.push("🟡 " + personel + " — " + gorevAdi + " (bugün son gün!)");
      }
    }
    if (gecikmisList.length > 0 || bugunList.length > 0) {
      var mesaj = "⚠️ *GÖREV UYARISI* — " + Utilities.formatDate(new Date(), "Europe/Istanbul", "dd.MM.yyyy") + "\n\n";
      if (bugunList.length > 0) mesaj += "*Bugün Biten Görevler:*\n" + bugunList.join("\n") + "\n\n";
      if (gecikmisList.length > 0) mesaj += "*Gecikmiş Görevler:*\n" + gecikmisList.join("\n");
      chatMesajGonder(mesaj, "mudur");
      chatMesajGonder(mesaj, "grup");
    }
  } catch(e) { Logger.log("Gecikme kontrol hata: " + e.message); }
}

function parasutTokenYenile_() {
  var ss = SpreadsheetApp.openById(CRM_SHEET_ID);
  var sh = ss.getSheetByName("PARASUT_AYAR");
  var data = sh.getDataRange().getValues();
  var ayar = {};
  for (var i = 0; i < data.length; i++) {
    ayar[String(data[i][0]||"").trim()] = String(data[i][1]||"").trim();
  }

  var response = UrlFetchApp.fetch("https://api.parasut.com/oauth/token", {
    method: "POST",
    contentType: "application/x-www-form-urlencoded",
    payload: {
      grant_type: "password",
      client_id: ayar["PARASUT_CLIENT_ID"],
      client_secret: ayar["PARASUT_CLIENT_SECRET"],
      username: ayar["PARASUT_USERNAME"],
      password: ayar["PARASUT_PASSWORD"],
      redirect_uri: "urn:ietf:wg:oauth:2.0:oob"
    },
    muteHttpExceptions: true
  });

  var json = JSON.parse(response.getContentText());
  if (!json.access_token) throw new Error("Token alınamadı: " + response.getContentText());

  // Token'ı PARASUT_AYAR'a kaydet
  for (var i = 1; i < data.length; i++) {
    if (String(data[i][0]).trim() === "PARASUT_ACCESS_TOKEN") {
      sh.getRange(i + 1, 2).setValue(json.access_token);
    }
    if (String(data[i][0]).trim() === "PARASUT_REFRESH_TOKEN") {
      sh.getRange(i + 1, 2).setValue(json.refresh_token || "");
    }
  }

  return json.access_token;
}

function eksiksizYedekle() {
  try {
    var scriptId = ScriptApp.getScriptId();
    var token = ScriptApp.getOAuthToken();
    
    var url = "https://script.googleapis.com/v1/projects/" + scriptId + "/content";
    var response = UrlFetchApp.fetch(url, {
      headers: { "Authorization": "Bearer " + token },
      muteHttpExceptions: true
    });
    
    if (response.getResponseCode() !== 200) {
      Logger.log("API hatası: " + response.getResponseCode());
      return { ok: false, msg: "API hatası: " + response.getResponseCode() };
    }
    
    var data = JSON.parse(response.getContentText());
    var files = data.files || [];
    
    var klasorAdi = "AVRUPABIR CRM YEDEK";
    var klasorler = DriveApp.getFoldersByName(klasorAdi);
    var klasor = klasorler.hasNext() ? klasorler.next() : DriveApp.createFolder(klasorAdi);
    
    var tarihStr = Utilities.formatDate(new Date(), "Europe/Istanbul", "dd-MM-yyyy_HH-mm");
    var tarihKlasor = klasor.createFolder("YEDEK_" + tarihStr);
    
    var kaydedilen = [];
    Logger.log("Toplam " + files.length + " dosya yedeklenecek...");
    
    for (var i = 0; i < files.length; i++) {
      var file = files[i];
      var dosyaAdi = file.name;
      var icerik = file.source || "";
      var tip = file.type || "SERVER_JS";
      
      var uzanti = ".gs";
      if (tip === "HTML") uzanti = ".html";
      if (tip === "JSON") uzanti = ".json";
      
      var blob = Utilities.newBlob(icerik, "text/plain", dosyaAdi + uzanti);
      tarihKlasor.createFile(blob);
      kaydedilen.push(dosyaAdi + uzanti);
      Logger.log((i+1) + "/" + files.length + " — " + dosyaAdi + uzanti + " (" + icerik.length + " karakter)");
    }
    
    var ozet = "AVRUPABIR CRM YEDEK OZETI\n";
    ozet += "Tarih: " + Utilities.formatDate(new Date(), "Europe/Istanbul", "dd.MM.yyyy HH:mm") + "\n";
    ozet += "Toplam dosya: " + kaydedilen.length + "\n\n";
    kaydedilen.forEach(function(d, idx) { ozet += (idx+1) + ". " + d + "\n"; });
    tarihKlasor.createFile(Utilities.newBlob(ozet, "text/plain", "OZET.txt"));
    
    var msg = "✅ " + kaydedilen.length + " dosya yedeklendi → " + klasorAdi + "/YEDEK_" + tarihStr;
    Logger.log(msg);
    return { ok: true, msg: msg, dosyalar: kaydedilen };
    
  } catch(e) {
    Logger.log("Yedekleme hatası: " + e.message);
    return { ok: false, msg: e.message };
  }
}

function _mahsupTarihFormat(deger) {
  if (!deger) return "";
  if (deger instanceof Date) {
    return Utilities.formatDate(deger, "Europe/Istanbul", "dd.MM.yyyy");
  }
  var s = String(deger).trim();
  // ISO format: 2026-03-04
  if (s.match(/^\d{4}-\d{2}-\d{2}/)) {
    var parts = s.substring(0, 10).split("-");
    return parts[2] + "." + parts[1] + "." + parts[0];
  }
  return s;
}
