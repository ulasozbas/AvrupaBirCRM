<script>
  function yukle() {
    document.getElementById('guncelleme-zaman').textContent = 'Yükleniyor...';
    google.script.run
      .withSuccessHandler(render)
      .withFailureHandler(hataGoster)
      .dashboardVerisiAl();
  }

  function render(veri) {
    document.getElementById('guncelleme-zaman').textContent = 'Son güncelleme: ' + veri.guncelleme;

    // ── 4 Kart ──
    const k = veri.kartlar;
    document.getElementById('kartlar').innerHTML = `
      <div class="kart yesil">
        <div class="etiket">🏦 Kasa Nakit</div>
        <div class="tutar">${formatPara(k.kasaNakit)}</div>
        <div class="alt-bilgi">Anlık nakit bakiye</div>
      </div>
      <div class="kart mavi">
        <div class="etiket">🏛️ Banka Nakit</div>
        <div class="tutar">${formatPara(k.bankaNakit)}</div>
        <div class="alt-bilgi">Tüm banka hesapları toplamı</div>
      </div>
      <div class="kart turuncu">
        <div class="etiket">⚠️ Toplam Borçlar</div>
        <div class="tutar">${formatPara(k.toplamBorclar)}</div>
        <div class="alt-bilgi">Çek + SGK + Vergi bekleyenler</div>
      </div>
      <div class="kart mor">
        <div class="etiket">📌 Aylık Sabit Gider</div>
        <div class="tutar">${formatPara(k.aylikSabitGider)}</div>
        <div class="alt-bilgi">Planlanan toplam / ay</div>
      </div>
    `;

    // ── Skorboard ──
    const s = veri.skorboard;
    const genislik = Math.min(s.yuzde, 100);
    document.getElementById('skorboard').innerHTML = `
      <div class="skorboard-baslik">
        <div>
          <h3>🎯 Aylık Ödeme Skorboard'u — ${s.donem}</h3>
          <div style="font-size:11px;color:#636e72;margin-top:2px">
            Sabit giderler için gereken ${formatPara(s.hedefTutar)}'e karşı gerçekleşen tahsilat
          </div>
        </div>
        <div class="yuzde-badge ${s.durum}">%${s.yuzde}</div>
      </div>
      <div class="progress-bar">
        <div class="progress-dolgu ${s.durum}" style="width:${genislik}%"></div>
      </div>
      <div class="skor-detay">
        <span>Giren: <strong>${formatPara(s.gerceklesen)}</strong></span>
        <span>Hedef: <strong>${formatPara(s.hedefTutar)}</strong></span>
        <span>Kalan: <strong>${formatPara(s.kalan)}</strong></span>
      </div>
      <div class="skor-mesaj">${s.mesaj}</div>
    `;

    // ── Son Kasa Hareketleri ──
    const kasaListesi = document.getElementById('son-kasa');
    if (veri.sonKasaHareketler && veri.sonKasaHareketler.length > 0) {
      kasaListesi.innerHTML = veri.sonKasaHareketler.map(h => `
        <li>
          <span class="h-tarih">${h.tarih}</span>
          <span class="h-aciklama">${h.aciklama}</span>
          <span class="h-tutar ${h.tip === 'Giriş' ? 'giris' : 'cikis'}">
            ${h.tip === 'Giriş' ? '+' : '-'}${formatPara(h.tutar)}
          </span>
        </li>
      `).join('');
    } else {
      kasaListesi.innerHTML = '<li><span class="h-aciklama" style="color:#b2bec3">Hareket yok</span></li>';
    }

    // ── Son Tahsilatlar ──
    const tahsilatListesi = document.getElementById('son-tahsilat');
    if (veri.sonTahsilatlar && veri.sonTahsilatlar.length > 0) {
      tahsilatListesi.innerHTML = veri.sonTahsilatlar.map(t => `
        <li>
          <span class="h-tarih">${t.tarih}</span>
          <span class="h-aciklama">${t.musteriAd}</span>
          <span class="badge ${odemeTipBadge(t.odemeTipi)}">${odemeTipKisa(t.odemeTipi)}</span>
          <span class="h-tutar giris" style="margin-left:6px">+${formatPara(t.tutar)}</span>
        </li>
      `).join('');
    } else {
      tahsilatListesi.innerHTML = '<li><span class="h-aciklama" style="color:#b2bec3">Tahsilat yok</span></li>';
    }

    // ── Tahsilat Dağılımı ──
    const at = veri.aylikTahsilat;
    document.getElementById('tahsilat-dagilim').innerHTML = `
      <div style="text-align:center;min-width:80px">
        <div style="font-size:16px;font-weight:700;color:#00b894">${formatPara(at.nakit)}</div>
        <div class="badge nakit" style="margin-top:3px">Nakit</div>
      </div>
      <div style="text-align:center;min-width:80px">
        <div style="font-size:16px;font-weight:700;color:#0984e3">${formatPara(at.kart)}</div>
        <div class="badge kart" style="margin-top:3px">K.Kartı</div>
      </div>
      <div style="text-align:center;min-width:80px">
        <div style="font-size:16px;font-weight:700;color:#e17055">${formatPara(at.eft)}</div>
        <div class="badge eft" style="margin-top:3px">EFT</div>
      </div>
      <div style="text-align:center;min-width:80px">
        <div style="font-size:16px;font-weight:700;color:#6c5ce7">${formatPara(at.cari)}</div>
        <div class="badge cari" style="margin-top:3px">Cari</div>
      </div>
      <div style="text-align:center;min-width:80px;margin-left:auto">
        <div style="font-size:16px;font-weight:700">${formatPara(at.toplamTutar)}</div>
        <div style="font-size:11px;color:#636e72;margin-top:3px">Toplam (${at.adet} işlem)</div>
      </div>
    `;
  }

  function hataGoster(hata) {
    document.getElementById('kartlar').innerHTML = `<div class="hata" style="grid-column:1/-1">Hata: ${hata.message}</div>`;
  }

  function formatPara(tutar) {
    if (tutar === undefined || tutar === null || isNaN(tutar)) return '₺0,00';
    return '₺' + Number(tutar).toLocaleString('tr-TR', {minimumFractionDigits: 2, maximumFractionDigits: 2});
  }

  function odemeTipBadge(tip) {
    if (tip === 'Nakit') return 'nakit';
    if (tip === 'Kredi Kartı (POS)') return 'kart';
    if (tip === 'Havale / EFT') return 'eft';
    if (tip === 'Cari Hesap') return 'cari';
    return '';
  }

  function odemeTipKisa(tip) {
    if (tip === 'Nakit') return 'Nakit';
    if (tip === 'Kredi Kartı (POS)') return 'K.Kartı';
    if (tip === 'Havale / EFT') return 'EFT';
    if (tip === 'Cari Hesap') return 'Cari';
    return tip;
  }

  // Sayfa açılınca yükle
  yukle();
  // Her 60 saniyede otomatik yenile
  setInterval(yukle, 60000);
</script>
