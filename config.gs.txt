/**
 * Hizmet tipine göre KDV oranını döner.
 * Tanımlı değilse AYAR sheet'teki varsayılan oran kullanılır.
 *
 * @param {string} hizmetTipi
 * @returns {number} KDV oranı (10 veya 20)
 */
function kdvOraniAl(hizmetTipi) {
  if (KDV_ORANLARI[hizmetTipi] !== undefined) {
    return KDV_ORANLARI[hizmetTipi];
  }
  // Tanımlı değilse AYAR'daki varsayılana bak, o da yoksa 20
  const varsayilan = parseFloat(ayarOku('KDV_VARSAYILAN') || '20');
  Logger.log(`⚠️ "${hizmetTipi}" için KDV oranı tanımlı değil, varsayılan %${varsayilan} kullanılıyor.`);
  return varsayilan;
}

/**
 * Hizmet tipine ve net tutara göre KDV tutarını hesaplar.
 *
 * @param {string} hizmetTipi
 * @param {number} netTutar    - KDV hariç tutar
 * @returns {{ kdvOrani, kdvTutar, toplam }}
 */
function kdvHesapla(hizmetTipi, netTutar) {
  const kdvOrani = kdvOraniAl(hizmetTipi);
  const kdvTutar = netTutar * (kdvOrani / 100);
  return {
    kdvOrani,
    kdvTutar,
    toplam: netTutar + kdvTutar,
  };
}

/**
 * Spreadsheet nesnesini döner.
 */
function getSpreadsheet() {
  return SpreadsheetApp.getActiveSpreadsheet();
}

/**
 * Sheet adına göre sheet nesnesini döner. Yoksa hata fırlatır.
 */
function getSheet(sheetAdi) {
  const ss = getSpreadsheet();
  const sheet = ss.getSheetByName(sheetAdi);
  if (!sheet) throw new Error(`"${sheetAdi}" sheet bulunamadı. Lütfen önce kurulum (setup) yapın.`);
  return sheet;
}

/**
 * Bugünün tarihini GG.AA.YYYY formatında döner.
 */
function bugun() {
  return Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'dd.MM.yyyy');
}

/**
 * Tarih nesnesini GG.AA.YYYY formatına çevirir.
 */
function tarihFormat(date) {
  if (!date) return '';
  if (typeof date === 'string') return date;
  return Utilities.formatDate(date, Session.getScriptTimeZone(), 'dd.MM.yyyy');
}

/**
 * Otomatik ID üretir. Prefix + 4 haneli numara (K-0001, B-0001 vb.)
 * Sheet'in son satırına bakarak bir sonraki numarayı verir.
 */
function yeniID(prefix, sheet) {
  const sonSatir = sheet.getLastRow();
  if (sonSatir <= 1) return `${prefix}-0001`; // Sadece başlık satırı var
  const mevcutID = sheet.getRange(sonSatir, 1).getValue().toString();
  const numara = parseInt(mevcutID.split('-')[1] || '0') + 1;
  return `${prefix}-${numara.toString().padStart(4, '0')}`;
}

/**
 * Tutarı Türk lirası formatında gösterir: 1.250,00 ₺
 */
function paraBirim(tutar) {
  if (isNaN(tutar)) return '0,00 ₺';
  return tutar.toLocaleString('tr-TR', { minimumFractionDigits: 2, maximumFractionDigits: 2 }) + ' ₺';
}

/**
 * Şu an giriş yapmış kullanıcının e-postasını döner.
 */
function aktifKullanici() {
  try {
    return Session.getActiveUser().getEmail() || 'sistem';
  } catch(e) {
    return 'sistem';
  }
}

/**
 * Bir sheet'e başlık satırı yazar (varsa dokunmaz).
 */
function baslikYaz(sheet, basliklar) {
  if (sheet.getLastRow() === 0) {
    const range = sheet.getRange(1, 1, 1, basliklar.length);
    range.setValues([basliklar]);
    range.setFontWeight('bold');
    range.setBackground('#f3f3f3');
    sheet.setFrozenRows(1);
  }
}

/**

    ['KDV_ISGUV',     '20',                         'İş güvenliği hizmetleri KDV oranı'],
    ['PARA_BIRIMI',   'TRY',                        'Para birimi'],
    ['DONEM',         '',                           'Aktif dönem (YYYY-MM, boş = otomatik)'],
  ];

  varsayilanlar.forEach(([anahtar, deger, aciklama]) => {
    if (!mevcutAyarlar.includes(anahtar)) {
      sheet.appendRow([anahtar, deger, aciklama]);
    }
  });
}

/**
 * AYAR sheet'ten bir değer okur.
 */
function ayarOku(anahtar) {
  try {
    const sheet = getSheet(SHEETS.AYAR);
    const data = sheet.getDataRange().getValues();
    for (let i = 1; i < data.length; i++) {
      if (data[i][0] === anahtar) return data[i][1];
    }
    return null;
  } catch(e) {
    return null;
  }
}

/**
 * AYAR sheet'e bir değer yazar / günceller.
 */
function ayarYaz(anahtar, deger) {
  const sheet = getSheet(SHEETS.AYAR);
  const data = sheet.getDataRange().getValues();
  for (let i = 1; i < data.length; i++) {
    if (data[i][0] === anahtar) {
      sheet.getRange(i + 1, 2).setValue(deger);
      return;
    }
  }
  sheet.appendRow([anahtar, deger, '']);
}
